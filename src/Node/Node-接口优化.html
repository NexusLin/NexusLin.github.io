优化分页排除 ​

之前我们的分页页码和页条条件是这样的：

js
// 页码数和页条数
function addPagination(query, params, pageNum, pageSize) {
    if (pageNum !== undefined && pageSize !== '' && pageSize !== null) {
        query += ' LIMIT ?, ?';
        let offset = (pageNum - 1) * pageSize;
        params.push(offset);
        params.push(parseInt(pageSize));
    }
    return query;
}

优化掉sql查询这种情况报错

JS
//错误查询
SELECT * FROM user WHERE name = '管理员' LIMIT '管理员', 0

//正确查询
SELECT * FROM user WHERE name = '管理员' LIMIT 0, 1;
js
// 页码数和页条数
function addPagination(query, params, pageNum, pageSize) {
    // 判断 pageNum 和 pageSize 是否为有效的数字
    if (pageNum !== undefined && !isNaN(pageNum) && pageNum > 0 && pageSize !== undefined && !isNaN(pageSize) && pageSize > 0) {
        query += ' LIMIT ?, ?';
        let offset = (pageNum - 1) * pageSize;
        params.push(offset);
        params.push(parseInt(pageSize));
    } else {
        // 如果无效，去除 LIMIT 子句，或者你也可以在这里设置默认值
        query = query.replace(/ LIMIT \?, \?/, '');
    }
    return query;
}

js
// 拼接查询条件
function addCondition(query, params, condition, value) {
    if (value !== undefined && value !== '' && value !== null) {
        query += params.length ? ' AND' : ' WHERE';
        query += ` ${condition} = ?`;
        params.push(value);
    }
    return query;
}

// 查询接口
function listApi(queryData, sqltotal, req, res, querySearch) {
    let querySearchData = querySearch;
    // 查询Sql
    let querySql = queryData;
    // 构建查询条件
    let params = [];
    Object.keys(querySearchData).forEach(key => {
        if (key != 'page_num' && key != 'page_size') {
            querySql = addCondition(querySql, params, key, querySearchData[key]);
        }
    });
    if (querySearchData['page_num'] && querySearchData['page_size']) {
        querySql = addPagination(querySql, params, querySearchData.page_num, querySearchData.page_size); //分页条件 
    }
    // 查询数据库并返回数据
    connectionPool.query(querySql, params, (err, results) => {
        if (err) {
            res.send({
                code: 500,
                message: '添加失败！'
            });
            return;
        } else {
            if (results.length > 0) {
                // 查询数据库并返回数据
                connectionPool.query(sqltotal, (errtotal, rows) => {
                    let total = rows[0]['total'];
                    if (errtotal) {
                        res.send({
                            code: 401,
                            message: '查询失败！'
                        });
                        return;
                    } else {
                        console.log('查询成功');
                        res.send({
                            total: total,
                            code: 200,
                            data: convertToCamelCase(results),
                            message: '查询成功！',
                        });
                    }
                });
            } else {
                res.send({
                    total: 0,
                    code: 200,
                    data: [],
                    message: '查询成功！',
                });
            }
        }
    });
}
