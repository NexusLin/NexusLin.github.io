method:GET

// params(请求参数)
{
   pageNum:1, // 页数 
   pageSize:10,// 条数
   dictName:'字典名称', // 查询参数-字典名称
   dictType:'字典类型', // 查询参数-字典类型
   status:'状态' // 查询参数-状态
   beginTime:'开始时间', // 查询参数-开始时间
   endTime:'结束时间' // 查询参数-结束时间

// 返回参数

    createTime: "2024-05-28 05:43:40", // 创建时间

    dictId: 1, // 字典id
    dictType: "sys_user_sex", // 字典类型
    remark: "用户性别列表", // 备注

}
