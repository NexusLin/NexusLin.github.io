NestJS-文章数据模块 ​

1、文章数据模块 ​

接下来我们新建一个文章数据模块，实现对于文章数据部分的管理功能接口

👉生成文章数据文件 ​

javascript
nest g controller modules/article --no-spec
nest g module modules/article --no-spec
nest g service modules/article --no-spec

创建完成以后的目录如下图所示

javascript
src\modules\article
  src\modules\article\article.controller.ts
  src\modules\article\article.module.ts
  src\modules\article\article.service.ts

//手动添加实体
  src\modules\article\article.entity.ts

👉article.entity.ts实体搭建 ​

javascript
import { Entity, PrimaryGeneratedColumn, Column } from 'typeorm';

@Entity('sys_article')
export class Article {
  @PrimaryGeneratedColumn({ name: 'article_id', type: 'int' })
  articleId: number;

  @Column({ name: 'title', length: 255, nullable: false })
  title: string;

  @Column({ name: 'author', type: 'varchar', length: 255, nullable: true, default: null })
  author: string | null;

  @Column({ name: 'type', type: 'varchar', length: 50, nullable: true, default: null })
  type: string | null;

  @Column({ name: 'articleclick', type: 'int', nullable: true, default: null })
  articleclick: number | null;

  @Column({ 
    name: 'status', 
    type: 'int', 
    nullable: true, 
    default: null, 
    comment: '0 暂存 1 发布' 
  })
  status: number | null;

  @Column({ name: 'articlecollect', type: 'int', nullable: true, default: null })
  articlecollect: number | null;

  @Column({ name: 'content', type: 'text', nullable: true, default: null })
  content: string | null;

  @Column({ name: 'thumburl', type: 'longtext', nullable: true, default: null })
  thumburl: string | null;

  @Column({ name: 'articlepublishTime', type: 'datetime', nullable: true, default: null })
  articlepublishTime: Date | null;

  @Column({ name: 'author_id', type: 'int', nullable: true, default: null })
  authorId: number | null;

  @Column({ name: 'create_time', type: 'datetime', nullable: true, default: () => 'CURRENT_TIMESTAMP' })
  createTime: Date;

  @Column({ name: 'update_time', type: 'datetime', nullable: true, default: () => 'CURRENT_TIMESTAMP', onUpdate: 'CURRENT_TIMESTAMP' })
  updateTime: Date;

  @Column({ name: 'uploadfile', type: 'longtext', nullable: true, default: null })
  uploadfile: string | null;

  @Column({ 
    type: 'boolean', 
    nullable: true, 
    default: false, 
    comment: '是否软删除消息（默认为0 未删除 1 删除）'
  })
  isDeleted: boolean;
}

👉实体映射 ​

在我们的src\common\entity\index.ts文件之中导入实体并进行映射

javascript
import { article } from '@/modules/article/article.entity'; // 文章数据类型实体

export const dataOrmConfig: dataOrmModuleOptions = {
  data: 'mysql', // 可以选择其他数据库，如 MySQL
  host: 'localhost',
  port: 3306,
  username: 'root',
  password: 'XXXXX',
  database: 'nexusnest',
  entities: [article], 
  synchronize: true, // 自动同步实体到数据库（开发时可以启用，生产环境应谨慎使用）
  logging: false,  // 启用日志记录
  timezone: '+08:00', // 设置时区
};

这个时候刷新我们数据库可以发现，数据库的表已经对应上我们对应的实体了

👉 article.module.ts ​

在文章数据模块之中导入其他的模块部分

javascript
import { Module } from '@nestjs/common';
import { TypeOrmModule } from '@nestjs/typeorm';

import { ArticleController } from './article.controller';
import { ArticleService } from './article.service';
import { Article } from './article.entity';

@Module({
  // 导入实体
  imports: [
    TypeOrmModule.forFeature([Article]),
  ],
  providers: [ArticleService],
  controllers: [ArticleController],
  exports: [ArticleService],
})
export class ArticleModule {}
