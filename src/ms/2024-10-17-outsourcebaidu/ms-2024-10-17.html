async function async1() {

    await async2();  // 进入async2，同时async2以后的函数等待他执行完成

    await async3(); //完成以后继续执行
}

async function async2() {
    console.log( 'async2' ) // 3 输出async2
}

new Promise(function(resolve) {
    console.log('promise 2'); // 4  输出promise2，同时后面的进入微任务

    console.log('promise4 then') // 10 then继续 输出'promise4 then'

分析以上代码

JavaScript的事件循环（event loop） 微任务（microtask） 宏任务（macrotask） 异步编程中的Promise和async/await

JavaScript的执行是基于事件循环的,包含了微任务和宏任务

微任务是在当前任务执行完毕后立即执行的任务, 而宏任务是在下一个事件循环中执行的任务。

在JavaScript中,Promise和async/await都是异步编程的常用

在上面的EventLoop执行顺序上,执行顺序是:同步任务->微任务->宏任务

上面从上到下先执行普通的同步任务，第一步输出 console.log('eventLoop');
//输出'eventLoop'

async1()函数调用,输出console.log('async1 start'); // 输出 'async1 start'

async1函数是异步的,碰到await时，会暂停该函数,直到后面的异步操作完成。所以执行 async2()函数,输出console.log('async2'); //输出 'async2'

接下来await后面的代码暂停着，因为await后面的代码是异步的，放入微任务队列1；这时候往下继续执行Promise，输出console.log('promise 2'); // 输出 'promise 2'

然后Promise.then()是微任务，所以放到微任务队列2 里，继续往下，第二个Promise也会同步执行，输出console.log('promise 4'); // 输出 'promise 4'

promise4 之后的放入微任务队列3里面

然后继续执行同步任务 console.log('eventLoop end'); // 输出console.log('eventLoop end');

然后执行微任务队列里的任务，按照上面我们放入的队列执行，这个时候'async1 end'开始

然后继续输出 // console.log( 'async1 end' ) 队列1

遇到async3，后面只有同步任务，于是 // 输出console.log('async3');

继续输出之前微任务 // console.log('promise2 then') 队列2 // console.log('promise4 then') 队列3

然后开始进入下一个事件循环依次输出 console.log('setTimeout 1'); //输出 console.log('promise 1'); //输出 console.log('promise then') //输出

    <script>
    // document.addEventListener('DOMContentLoaded', function() {});
    let notes = [];

    const addNoteButton = document.getElementById('addNoteButton');
    const saveNoteButton = document.getElementById('saveNoteButton');
    const closeModalButton = document.getElementById('closeModalButton');
    const noteContent = document.getElementById('noteContent');

    // 添加笔记按钮事件

        noteContent.value = ''; // 清空文本框
        document.getElementById('modalTitle').textContent = '添加笔记'; // 设置弹窗标题
        showModal();

    // 保存笔记按钮事件
    saveNoteButton.addEventListener('click', function() {
        const content = noteContent.value.trim();
        if (content) {
            if (currentNoteId) {
                // 编辑笔记
                updateNote(currentNoteId, content);
            } else {
                // 添加新笔记
                addNote(content);
            }
            closeModal();
        } else {
            alert('请写下你的笔记。');
        }

    function showModal() {
        modalOverlay.style.display = 'block';
        noteModal.style.display = 'block';
    }

    // 关闭弹窗
    function closeModal() {
        modalOverlay.style.display = 'none';
        noteModal.style.display = 'none';

    function addNote(content) {
        const noteId = Date.now().toString();
        const note = { id: noteId, content };
        saveNoteToLocalStorage(note);
        renderNote(note);

    // 更新笔记
    function updateNote(noteId, newContent) {
        const notes = getNotesFromLocalStorage();
        const noteIndex = notes.findIndex(note => note.id === noteId);
        if (noteIndex !== -1) {
            notes[noteIndex].content = newContent;
            localStorage.setItem('notes', JSON.stringify(notes));
            loadNotes(); // 重新加载笔记
        }
    }

    // 渲染笔记
    function renderNote(note) {
        const noteItem = document.createElement('div');
        noteItem.classList.add('note-item');
        noteItem.id = note.id;

        const noteText = document.createElement('div');
        noteText.classList.add('note-item-left');
        noteText.textContent = note.content;

        noteActions.classList.add('note-item-right');
        noteActions.classList.add('btnbox');

    // 删除笔记
    function deleteNote(noteId) {
        const notes = getNotesFromLocalStorage();
        const updatedNotes = notes.filter(note => note.id !== noteId);
        localStorage.setItem('notes', JSON.stringify(updatedNotes));
        loadNotes(); // 重新加载笔记
    }


    // 从本地存储加载笔记
    function loadNotes() {
        noteList.innerHTML = ''; // 清空当前列表
        notes = getNotesFromLocalStorage();
        notes.forEach(note => {
            renderNote(note);
        });
    }

    // 获取本地存储中的笔记
    function getNotesFromLocalStorage() {
        const notes = localStorage.getItem('notes');
        return notes ? JSON.parse(notes) : [];
    }

    // 保存笔记到本地存储
    function saveNoteToLocalStorage(note) {
        notes = getNotesFromLocalStorage();
        notes.push(note);
        localStorage.setItem('notes', JSON.stringify(notes));
    }


    // 查询笔记
    function queryNote() {
        const searchValue = document.getElementById('search').value.trim().toLowerCase();
        console.log(searchValue, '搜索内容');
        console.log(notes, 'notes');
        const filteredNotes = notes.filter(item => item.content.trim().toLowerCase().includes(searchValue));
        console.log(filteredNotes, 'filteredNotes');

        const noteList = document.getElementById('noteList');
        noteList.innerHTML = '';

        filteredNotes.forEach(note => {
            renderNote(note);
        });
    }
    </script>

3、第三题 ​

在线网盘 实现通过 Web 管理本地文件夹的功能，需求如下：

  • 配置将某个本地文件夹作为主目录。
  • 前端展现可以类似网盘（或者 Windows 资源管理器及 Mac Finder 等），可以修改文件名、创建文件夹、上传文件、删除文件等。
  • 更多功能可以自由发挥，包括展现方面可以自己设计。
  • 可以思考一下如何更好的管理文件？是否有更好的信息展现方式？

接下来我们就写一个通过 Web 管理本地文件夹的纯html应用，配置将某个本地文件夹作为主目录，可以修改文件名、创建文件夹、上传文件、删除文件等，并实现类似网盘一样的简约功能

js

<!DOCTYPE html>
<html lang="en">

<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Web File Manager</title>
    <style>
    body {
        font-family: Arial, sans-serif;
        margin: 20px;
        padding: 20px;
        border: 1px solid #ccc;
    }

    #fileList {
        margin-top: 20px;
    }

    .file-item {
        display: flex;
        justify-content: space-between;
        margin: 5px 0;
    }

    button {
        margin-left: 10px;
    }
    </style>
</head>

<body>
    <h1>Web File Manager</h1>
    <button id="chooseFolder">选择文件夹</button>
    <button id="createFolder">创建文件夹</button>
    <button id="uploadFile">上传文件</button>
    <div id="fileList"></div>
    <script>
    let currentDirectoryHandle;

    document.getElementById('chooseFolder').addEventListener('click', async () => {
        const options = {
            // 允许用户选择一个文件夹
            types: [{
                description: 'Directories',
                accept: { 'application/x-directory': ['*'] },
            }],
            excludeAcceptAllOption: true,
            multiple: false
        };
        currentDirectoryHandle = await window.showDirectoryPicker(options);
        updateFileList();
    });

    document.getElementById('createFolder').addEventListener('click', async () => {
        if (!currentDirectoryHandle) return;
        const folderName = prompt("请输入文件夹名称:");
        if (folderName) {
            await currentDirectoryHandle.getDirectoryHandle(folderName, { create: true });
            updateFileList();
        }
    });

    document.getElementById('uploadFile').addEventListener('click', async () => {
        if (!currentDirectoryHandle) return;
        const [fileHandle] = await window.showOpenFilePicker();
        await currentDirectoryHandle.getFileHandle(fileHandle.name, { create: true });
        const file = await fileHandle.getFile();
        const writable = await fileHandle.createWritable();
        await writable.write(file);
        await writable.close();
        updateFileList();
    });

    async function updateFileList() {
        if (!currentDirectoryHandle) return;
        const fileListDiv = document.getElementById('fileList');
        fileListDiv.innerHTML = '';

        for await (const entry of currentDirectoryHandle.values()) {
            const fileItem = document.createElement('div');
            fileItem.className = 'file-item';
            fileItem.textContent = entry.name;

            const deleteButton = document.createElement('button');
            deleteButton.textContent = '删除';
            deleteButton.onclick = async () => {
                await currentDirectoryHandle.removeEntry(entry.name);
                updateFileList();
            };

            fileItem.appendChild(deleteButton);
            fileListDiv.appendChild(fileItem);
        }
    }
    </script>
</body>

</html>
